#!/bin/bash
echo "==========================="
echo "Анализ дискового пространства"
echo "==========================="
echo ""
printf "%-20s %-10s\n" "Файловая система" "Использовано %"
echo "---------------------------"
df -h | awk 'NR>1 {
gsub(/%/, "", $5)
printf "%-20s %-10s%%\n", $1, $5
if ($5 > 90) {
print "ПРЕДУПРЕЖДЕНИЕ: Раздел " $1 " заполнен на " $5 "%!"
}
}'
